    int left = 2 * root + 1;  
    int right = 2 * root + 2;

    if (left < size && heap[left].value < heap[smallest].value)

void heapify(MinHeapNode heap[], int size, int root) {
    int smallest = root;
    int left = 2 * root + 1;
    int right = 2 * root + 2;

    if (left < size && heap[left].value < heap[smallest].value)
        smallest = left;
    if (right < size && heap[right].value < heap[smallest].value)
        smallest = right;

    if (smallest != root) {
        swap(heap[root], heap[smallest]);
        heapify(heap, size, smallest);
    }
}
